#version 330 core
out vec4 FragColor;

in vec2 TexCoords;
in vec3 ANormal;
in vec3 FragPos;
in vec3 ViewPos;

uniform sampler2D texture_diffuse1;
void main()
{    
	// Ambient Shading
    //FragColor = (vec4(1.0f, 1.0f, 0.0f, 0.0f)) * texture(texture_diffuse1, TexCoords);

	vec3 lightColor = vec3(1.0f, 1.0f, 1.0f);
	float ambientPower = 0.1;
	vec3 ambient = ambientPower * lightColor;

	vec3 lightPos = vec3(0.0f, 0.0f, 3.0f);
	vec3 norm = normalize(ANormal);
	vec3 lightDir = normalize(lightPos - FragPos);
	float diff = max(dot(norm, lightDir), 0.0);
	vec3 diffuse = diff * lightColor;

	float specularStrength = 1.5;
	vec3 viewDir = normalize(ViewPos - FragPos);
	vec3 reflectDir = reflect(-lightDir, norm);
	float spec = pow(max(dot(viewDir, reflectDir), 0.0), 32);
	vec3 specular = specularStrength * spec * lightColor;

	FragColor = (vec4(ambient, 1.0f) + vec4(diffuse, 1.0f) + vec4(specular, 1.0f)) * texture(texture_diffuse1, TexCoords);

	//FragColor = (vec4(ambient, 1.0f)) * texture(texture_diffuse1, TexCoords);
}
